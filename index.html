<!-- Put script here: BOTTOM of body -->

<script>

    function loadVidaSdk(callback) {

        const script = document.createElement("script");

        script.src = "https://web-sdk.vida.id/1.0.5/sandbox/vida-web-sdk.js";

        script.onload = callback;

        script.onerror = () => alert("Failed to load VIDA SDK");

        document.head.appendChild(script);

    }


    function startLiveness() {

        const token = document.getElementById("token").value.trim();

        const signingKey = document.getElementById("signKey").value.trim();


        if (!token || !signingKey) {

            alert("Please input token + signing key");

            return;

        }


        loadVidaSdk(() => {

            if (!window.VidaSDK) {

                alert("VIDA SDK not loaded");

                return;

            }


            console.log("SDK Loaded, initializing...");


            window.VidaSDK.init({

                token,

                signingKey,

                elementId: "sdk-container",

                locale: "EN",

                partnerTransactionId: crypto.randomUUID(),


                onComplete: (data) => {

                    console.log("Liveness Completed:", data);

                    alert("Liveness Completed! Check console.");

                },


                onError: (error) => {

                    console.error("Error:", error);

                    alert("Error occurred. Check console.");

                }

            });

        });

    }


    // Attach click handler after functions defined

    document.getElementById("startBtn").addEventListener("click", startLiveness);

</script>
